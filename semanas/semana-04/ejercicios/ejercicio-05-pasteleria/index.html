// PSEUDOCÓDIGO DE LÓGICA DE CÁLCULO

FUNCIÓN calcularTotal()
    raciones = OBTENER_VALOR_INPUT("raciones")
    sinGluten = ESTADO_CHECKBOX("sin_gluten")
    
    PRECIO_POR_RACION = 3
    SUPLEMENTO_GLUTEN = 5
    
    total = raciones * PRECIO_POR_RACION
    
    SI sinGluten ES VERDADERO ENTONCES
        total = total + SUPLEMENTO_GLUTEN
    FIN SI
    
    MOSTRAR_EN_UI(total)
    REMOVER_CLASE("hidden", "resultado_container")
FIN FUNCIÓN
